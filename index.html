    <script src="https://raw.githack.com/AR-js-org/AR.js/master/aframe/build/aframe-ar.js"></script>
    <style>
        body { margin: 0; overflow: hidden; font-family: Arial, sans-serif; }
        #scanText { position: absolute; top: 10px; left: 50%; transform: translateX(-50%); color: #FFD700; font-size: 1.5em; opacity: 0.8; z-index: 10; }
    </style>
</head>
<body>
    <div id="scanText">Наведи на маркер</div>
    <a-scene embedded id="arScene" arjs="sourceType: webcam; detectionMode: mono; trackingMethod: best;">
        <a-marker id="marker" type="pattern" url="marker.patt">
            <a-video src="ar-video.mp4" autoplay loop="true" width="1" height="1" position="0 0 0" rotation="-90 0 0"></a-video>
        </a-marker>
        <a-entity camera></a-entity>
    </a-scene>

    <script>
        const arScene = document.getElementById('arScene');
        const marker = document.getElementById('marker');
        const scanText = document.getElementById('scanText');
        const video = document.querySelector('a-video');

        arScene.addEventListener('loaded', async () => {
            if (arScene.components['arjs']) {
                // Налаштування камери через getUserMedia
                const constraints = {
                    video: {
                        facingMode: 'environment', // Задня камера
                        width: { ideal: 1920 },   // Максимальна ширина (Full HD)
                        height: { ideal: 1080 },  // Максимальна висота
                        frameRate: { ideal: 30 }, // Стабільна частота кадрів
                        focusMode: 'continuous',  // Постійне фокусування (якщо підтримується)
                    }
                };

                try {
                    const stream = await navigator.mediaDevices.getUserMedia(constraints);
                    arScene.components['arjs'].initWebcam(stream); // Ініціалізація з кастомним потоком
                    console.log('Камера увімкнена з налаштуваннями:', constraints);
                } catch (error) {
                    console.error('Помилка доступу до камери:', error);
                }
            }
        });

        marker.addEventListener('markerFound', () => {
            scanText.style.display = 'none';
            video.components.material.material.map.image.play();
        });

        marker.addEventListener('markerLost', () => {
            scanText.style.display = 'block';
            video.components.material.material.map.image.pause();
        });
    </script>
</body>
